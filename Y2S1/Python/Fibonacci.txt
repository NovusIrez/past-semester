num = int(input())

num_list = []


def fibonacci(n):
	#complete the recursive function
	curr = num - n
	if n <= 0:
		return
	elif curr <= 1:
		print(curr)
		num_list.append(curr)
	else:
		calc = num_list[curr - 2] + num_list[curr - 1]
		print(calc)
		num_list.append(calc)
	n -= 1
	fibonacci(n)


fibonacci(num)
